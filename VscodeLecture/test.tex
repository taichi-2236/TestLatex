\section{ショートカットキー}
\subsection{基本エディタ操作}
\begin{enumerate}
  \item サイドバー表示／非表示（Ctrl＋B）
  \item エクスプローラー表示（Ctrl＋Shift＋E）
  \item ターミナル表示／非表示（Ctrl＋＠）
  \item 新規ファイル作成（Ctrl＋N）
  \item ファイル保存（Ctrl＋S）
  \item ファイルを閉じる（Ctrl＋W）
\end{enumerate}

\subsection{行操作}
\begin{enumerate}
  \item 行移動（Alt＋上下）
  \item 行選択（Ctrl＋L）
  \item 行複製（Shift＋Alt＋上下）
  \item 行削除（Ctrl＋Shift＋K）
  \item 行コメントアウト（Ctrl＋/）
\end{enumerate}

\subsection{ファイル内操作}
\begin{enumerate}
  \item ファイル内を検索（Ctrl＋F）
  \item 選択範囲に一致する範囲を全選択（Ctrl＋Shift＋L）
  \item ファイル内で置換（Ctrl＋H）
  \item カーソルを上下に拡張（Ctrl＋Alt＋上下）
  \item インデントを入れる／減らす（Ctrl＋］／［）
\end{enumerate}

\subsection{応用エディタ操作}
\begin{enumerate}
  \item ファイル名を入力して開く（Ctrl＋P）
  \item フォルダ内を検索（Ctrl＋Shift＋F）
  \item ファイルを再度開く（Ctrl＋Shift＋T）
  \item 新規ウィンドウの立ち上げ（Ctrl＋Shift＋N）
  \item ワークスペース切り替え（Ctrl＋R）
  \item 画面を拡大（Ctrl＋；）
  \item 画面を縮小（Ctrl＋－）
  \item 右端で折り返す／解除（Alt＋Z）
  \item 定義へ移動（F12）
  \item 参照へ移動（Shift＋F12）
  \item 保存済みのファイルを全て閉じる（Ctrl＋K→U）
\end{enumerate}
